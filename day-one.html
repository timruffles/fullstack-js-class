Bullet-proof objects in JS

var account = BankAccount(50);
account.deposit(5);
account.balance() // 5;

function BankAccount(balance) {
  var self = {};

  self.deposit = function(n) {
    if(balance + n < 0) {
      throw new Error("you don't have an overdraft!");
    }
    balance += n;
  }

  self.balance = function() {
    return balance;
  }

  return self;
}
